CREATE TABLE faculty (
faculty_name CHAR(25) NOT NULL,
faculty_information TEXT,
PRIMARY KEY (faculty_name)
)
COLLATE=utf8_unicode_ci
ENGINE=InnoDB;

CREATE TABLE major_program (
parent_faculty CHAR(25) NOT NULL,

major_description TEXT,

PRIMARY KEY (parent_faculty, major_name),
FOREIGN KEY (parent_faculty) REFERENCES faculty(faculty_name)
)
COLLATE=utf8_unicode_ci
ENGINE=InnoDB;

)
